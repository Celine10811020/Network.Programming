//開三個視窗
//1. server端
//建立一個Port 12345的連線
nc -l 12345
//2.
//只看Port 12345的連線
netstat -na | grep 12345
//看全部連線
netstat -na | less //0.0.0.0表示可以接受任何的連線
/*
只設server端→LISTEN
用戶端也連上後→ESTABLISHED
剛離開後→TIME_WAIT
離開一陣子後→消失不見
*/
//離開
q
//3. 用戶端
nc localhost 12345
//連線斷掉
^{c}
